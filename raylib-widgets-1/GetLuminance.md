/**
 * @brief Обчислює яскравість кольору (luminance) за формулою зваженого середнього.
 * 
 * Використовується стандартна формула, що враховує сприйняття людським оком різних кольорів:
 * - Червоний має найбільший вплив (0.2126)
 * - Зелений має найбільший внесок у сприйняття яскравості (0.7152)
 * - Синій має найменший внесок (0.0722)
 * 
 * Ця функція дозволяє визначити, наскільки "світлим" або "темним" є колір,
 * що важливо для вибору контрастного кольору тексту, щоб забезпечити хорошу читабельність.
 * 
 * @param color Колір у форматі raylib Color (r,g,b,a)
 * @return float Значення яскравості в діапазоні [0,1], де 0 — чорний, 1 — білий
 */
static float GetLuminance(Color color)
{
    float r = color.r / 255.0f;
    float g = color.g / 255.0f;
    float b = color.b / 255.0f;
    return 0.2126f * r + 0.7152f * g + 0.0722f * b;
}

/**
 * @brief Вибирає контрастний колір тексту (чорний або білий) залежно від яскравості фону.
 * 
 * Читабельність тексту залежить від контрастності між кольором тексту і фоном.
 * Якщо фон світлий (яскравість більше 0.5), текст робиться чорним для кращої видимості.
 * Якщо фон темний, текст робиться білим.
 * 
 * Це базовий підхід, що відповідає рекомендаціям з доступності (accessibility),
 * допомагає людям з порушеннями зору краще сприймати інформацію.
 * 
 * @param bgColor Колір фону
 * @return Color Чорний або білий колір для тексту, що контрастує з фоном
 */
static Color GetContrastColor(Color bgColor)
{
    return (GetLuminance(bgColor) > 0.5f) ? BLACK : WHITE;
}
